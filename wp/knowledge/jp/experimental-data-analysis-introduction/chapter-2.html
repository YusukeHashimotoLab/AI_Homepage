    """
    Asymmetric Least Squaresæ³•ã«ã‚ˆã‚‹ãƒ™ãƒ¼ã‚¹ãƒ©ã‚¤ãƒ³æ¨å®š

    

    
    Parameters:
    -----------

    lam : float
        å¹³æ»‘åŒ–ãƒ‘ãƒ©ãƒ¡ãƒ¼ã‚¿ï¼ˆå¤§ãã„ã»ã©æ»‘ã‚‰ã‹ï¼‰

    """
    L = len(y)
    D = sparse.diags([1, -2, 1], [0, -1, -2], shape=(L, L-2))
    w = np.ones(L)

        z = spsolve(Z, w * y)
        w = p  (y > z) + (1 - p)  (y < z)

    

    
    return z

    def adaptive_prominence(self, spectrum: np.ndarray) -> float:
