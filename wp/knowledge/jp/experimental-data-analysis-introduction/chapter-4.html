from sklearn.pipeline import Pipeline
from sklearn.impute import SimpleImputer
from sklearn.preprocessing import StandardScaler, RobustScaler
from sklearn.decomposition import PCA
from sklearn.cluster import KMeans
import joblib

class IntegratedAnalysisPipeline:
    """Áµ±ÂêàËß£Êûê„Éë„Ç§„Éó„É©„Ç§„É≥"""

    def __init__(self, n_clusters=3):
        """
        Parameters:
        -----------
        n_clusters : int
            „ÇØ„É©„Çπ„Çø„ÉºÊï∞Ôºà„Éó„É≠„Çª„ÇπÁä∂ÊÖã„ÅÆÊï∞Ôºâ
        """
        self.pipeline = Pipeline([
            ('imputer', SimpleImputer(strategy='median')),  # Ê¨†ÊêçÂÄ§Ë£úÂÆå
            ('scaler', RobustScaler()),  # Â§ñ„ÇåÂÄ§„Å´È†ëÂÅ•„Å™Ê®ôÊ∫ñÂåñ
            ('pca', PCA(n_components=0.95)),  # Á¥ØÁ©çÂØÑ‰∏éÁéá95%„Åæ„Åß
            ('clustering', KMeans(n_clusters=n_clusters, random_state=42))
        ])
        self.n_clusters = n_clusters

    def fit(self, X):
        """„Éë„Ç§„Éó„É©„Ç§„É≥Â≠¶Áøí"""
        self.pipeline.fit(X)
        return self

    def transform(self, X):
        """Ê¨°ÂÖÉÂâäÊ∏õ„ÅÆ„Åø"""
        # „ÇØ„É©„Çπ„Çø„É™„É≥„Ç∞Ââç„Åæ„Åß„ÅÆ„Çπ„ÉÜ„ÉÉ„Éó„ÇíÂÆüË°å
        X_transformed = X.copy()
        for step_name, step in self.pipeline.steps[:-1]:
            X_transformed = step.transform(X_transformed)
        return X_transformed

    def predict(self, X):
        """„ÇØ„É©„Çπ„Çø„Éº‰∫àÊ∏¨"""
        return self.pipeline.predict(X)

    def get_feature_importance(self, feature_names):
        """‰∏ªÊàêÂàÜ„Å´„Åä„Åë„ÇãÁâπÂæ¥Èáè„ÅÆÈáçË¶ÅÂ∫¶"""
        pca = self.pipeline.named_steps['pca']
        loadings = pca.components_.T * np.sqrt(pca.explained_variance_)

        importance_df = pd.DataFrame(
            loadings,
            columns=[f'PC{i+1}' for i in range(pca.n_components_)],
            index=feature_names
        )
        return importance_df

    def save(self, filename):
        """„É¢„Éá„É´‰øùÂ≠ò"""
        joblib.dump(self.pipeline, filename)

    @staticmethod
    def load(filename):
        """„É¢„Éá„É´Ë™≠„ÅøËæº„Åø"""
        return joblib.load(filename)
