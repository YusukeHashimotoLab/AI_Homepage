    """Li-ionÈõªÊ±†„ÅÆ„Ç§„Ç™„É≥‰ºùÂ∞éÂ∫¶Ôºà‰ªÆÊÉ≥ÁöÑ„Å™‰æãÔºâ"""

def ucb_acquisition(x, gp, kappa=2.0):
    """
    Upper Confidence BoundÁç≤ÂæóÈñ¢Êï∞

    Parameters:
    -----------
    x : array
        Ë©ï‰æ°ÁÇπ
    gp : GaussianProcessRegressor
        Â≠¶ÁøíÊ∏à„Åø„Ç¨„Ç¶„ÇπÈÅéÁ®ã„É¢„Éá„É´
    kappa : float
        Êé¢Á¥¢„ÅÆÂº∑„ÅïÔºàÂ§ß„Åç„ÅÑ„Åª„Å©Êé¢Á¥¢ÈáçË¶ñÔºâ
    """

    return mean + kappa * std

def bayesian_optimization_demo(n_iterations, initial_samples=3):
    """
    „Éô„Ç§„Ç∫ÊúÄÈÅ©Âåñ„ÅÆ„Éá„É¢„É≥„Çπ„Éà„É¨„Éº„Ç∑„Éß„É≥

    Parameters:
    -----------
    n_iterations : int
        ÊúÄÈÅ©Âåñ„ÅÆ„Ç§„ÉÜ„É¨„Éº„Ç∑„Éß„É≥Êï∞
    initial_samples : int
        ÂàùÊúü„É©„É≥„ÉÄ„É†„Çµ„É≥„Éó„É´Êï∞

    Returns:
    --------
    X_sampled : array
        „Çµ„É≥„Éó„É™„É≥„Ç∞„Åó„ÅüÁÇπ
    y_observed : array
        Ë¶≥Ê∏¨ÂÄ§
    """
    # ÂàùÊúü„É©„É≥„ÉÄ„É†„Çµ„É≥„Éó„É™„É≥„Ç∞
    X_sampled = np.random.uniform(0, 1, initial_samples)

    # „Ç¨„Ç¶„ÇπÈÅéÁ®ã„É¢„Éá„É´„ÅÆÂàùÊúüÂåñ
    kernel = ConstantKernel(1.0) * RBF(length_scale=0.1)
    gp = GaussianProcessRegressor(kernel=kernel, n_restarts_optimizer=10)

    # ÈÄêÊ¨°ÁöÑ„Çµ„É≥„Éó„É™„É≥„Ç∞
    for i in range(n_iterations - initial_samples):
        # „Ç¨„Ç¶„ÇπÈÅéÁ®ã„ÇíÂ≠¶Áøí
        gp.fit(X_sampled.reshape(-1, 1), y_observed)

        # Áç≤ÂæóÈñ¢Êï∞„ÇíÊúÄÂ§ßÂåñ„Åô„ÇãÁÇπ„ÇíÊé¢Á¥¢
        X_candidate = np.linspace(0, 1, 1000)
        acq_values = ucb_acquisition(X_candidate, gp)
        next_x = X_candidate[np.argmax(acq_values)]

        # Ê¨°„ÅÆÂÆüÈ®ì„ÇíÂÆüË°å
        next_y = objective(next_x)

        # „Éá„Éº„Çø„Å´ËøΩÂä†
        X_sampled = np.append(X_sampled, next_x)
        y_observed = np.append(y_observed, next_y)

    return X_sampled, y_observed

def electrolyte_conductivity(
    solvent_ratio,
    salt_concentration,
    additive_concentration
):
    """
