import numpy as np

def electrolyte_conductivity(
    solvent_ratio,
    salt_concentration,
    additive_concentration
):
    """
    é›»è§£è³ªã®ã‚¤ã‚ªãƒ³ä¼å°åº¦ã‚’è¨ˆç®—ï¼ˆç°¡ç•¥åŒ–ãƒ¢ãƒ‡ãƒ«ï¼‰

    Parameters:
    -----------
    solvent_ratio : float
        æœ‰æ©Ÿæº¶åª’ã®æ··åˆæ¯” (0-1)
    salt_concentration : float
        Liå¡©ã®æ¿ƒåº¦ (0.5-2.0 M)
    additive_concentration : float
        æ·»åŠ å‰¤ã®æ¿ƒåº¦ (0-5 wt%)

    Returns:
    --------
    float : ã‚¤ã‚ªãƒ³ä¼å°åº¦ (mS/cm)
    """
    # ç°¡ç•¥åŒ–ã•ã‚ŒãŸçµŒé¨“å¼ï¼ˆå®Ÿéš›ã¯ã‚ˆã‚Šè¤‡é›‘ï¼‰
    base_conductivity = 10.0

    # æº¶åª’åŠ¹æœï¼ˆæœ€é©æ¯”ã¯0.6ä»˜è¿‘ï¼‰
    solvent_effect = np.exp(-10 * (solvent_ratio - 0.6)**2)

    # å¡©æ¿ƒåº¦åŠ¹æœï¼ˆæœ€é©ã¯1.0 Mä»˜è¿‘ï¼‰
    salt_effect = salt_concentration * np.exp(-0.5 * (salt_concentration - 1.0)**2)

    # æ·»åŠ å‰¤åŠ¹æœï¼ˆå°‘é‡ã§åŠ¹æœã‚ã‚Šï¼‰
    additive_effect = 1 + 0.3 * np.exp(-additive_concentration / 2)

    # ãƒ©ãƒ³ãƒ€ãƒ ãƒã‚¤ã‚ºï¼ˆå®Ÿé¨“èª¤å·®ï¼‰
    noise = np.random.normal(0, 0.5)

    conductivity = (base_conductivity * solvent_effect *
                    salt_effect * additive_effect + noise)

    return max(0, conductivity)
