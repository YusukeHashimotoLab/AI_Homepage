import numpy as np
import matplotlib.pyplot as plt

def lennard_jones(r, epsilon=1.0, sigma=1.0):
    """Lennard-Jones„Éù„ÉÜ„É≥„Ç∑„É£„É´"""
    return 4 * epsilon * ((sigma/r)12 - (sigma/r)6)

def lj_force(r, epsilon=1.0, sigma=1.0):
    """Lennard-Jones„Éù„ÉÜ„É≥„Ç∑„É£„É´„ÅÆÂäõ"""
    return 24 * epsilon * (2*(sigma/r)13 - (sigma/r)7) / r

def velocity_verlet_md(N_steps=1000, dt=0.001):
    """
    Velocity VerletÊ≥ï„Å´„Çà„Çã1Ê¨°ÂÖÉMD„Ç∑„Éü„É•„É¨„Éº„Ç∑„Éß„É≥
    2„Å§„ÅÆLennard-JonesÁ≤íÂ≠ê
    """
    # ÂàùÊúüÊù°‰ª∂
    r = np.array([0.0, 2.0])  # ‰ΩçÁΩÆ
    v = np.array([0.5, -0.5])  # ÈÄüÂ∫¶
    m = np.array([1.0, 1.0])  # Ë≥™Èáè

    # Â±•Ê≠¥‰øùÂ≠òÁî®
    r_history = np.zeros((N_steps, 2))
    v_history = np.zeros((N_steps, 2))
    E_history = np.zeros(N_steps)

    for step in range(N_steps):
        # Âäõ„ÅÆË®àÁÆó
        r12 = r[1] - r[0]
        F = lj_force(abs(r12))
        a = np.array([-F/m[0], F/m[1]])  # Âä†ÈÄüÂ∫¶

        # ‰ΩçÁΩÆ„ÅÆÊõ¥Êñ∞
        r = r + v * dt + 0.5 * a * dt**2

        # Êñ∞„Åó„ÅÑÂäõ„ÅÆË®àÁÆó
        r12_new = r[1] - r[0]
        F_new = lj_force(abs(r12_new))
        a_new = np.array([-F_new/m[0], F_new/m[1]])

        # ÈÄüÂ∫¶„ÅÆÊõ¥Êñ∞
        v = v + 0.5 * (a + a_new) * dt

        # „Ç®„Éç„É´„ÇÆ„ÉºË®àÁÆó
        KE = 0.5 * np.sum(m * v**2)
        PE = lennard_jones(abs(r12_new))
        E_total = KE + PE

        # ‰øùÂ≠ò
        r_history[step] = r
        v_history[step] = v
        E_history[step] = E_total

    return r_history, v_history, E_history
