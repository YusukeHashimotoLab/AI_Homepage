class MolecularVAE(nn.Module):
    """
    Á∞°ÊòìÁâàÂàÜÂ≠êVAEÔºàSMILESÊñáÂ≠óÂàó„Éô„Éº„ÇπÔºâ

    „Ç¢„Éº„Ç≠„ÉÜ„ÇØ„ÉÅ„É£:
    - Encoder: SMILES ‚Üí ÊΩúÂú®„Éô„ÇØ„Éà„É´ÔºàÂπ≥ÂùáŒº„ÄÅÂàÜÊï£œÉ¬≤Ôºâ
    - Decoder: ÊΩúÂú®„Éô„ÇØ„Éà„É´ ‚Üí SMILES
    - ÊêçÂ§±: ÂÜçÊßãÊàêË™§Â∑Æ + KL„ÉÄ„Ç§„Éê„Éº„Ç∏„Çß„É≥„Çπ
    """

    def __init__(self, vocab_size=50, max_length=120, latent_dim=128):
        """
        Parameters:
        -----------
        vocab_size : int
            SMILESË™ûÂΩô„Çµ„Ç§„Ç∫ÔºàÊñáÂ≠óÁ®ÆÈ°ûÊï∞Ôºâ
        max_length : int
            ÊúÄÂ§ßSMILESÈï∑
        latent_dim : int
            ÊΩúÂú®Á©∫Èñì„ÅÆÊ¨°ÂÖÉÊï∞
        """
        super(MolecularVAE, self).__init__()

        self.vocab_size = vocab_size
        self.max_length = max_length
        self.latent_dim = latent_dim

        # EncoderÔºàSMILES ‚Üí ÊΩúÂú®„Éô„ÇØ„Éà„É´Ôºâ
        self.encoder_embedding = nn.Embedding(vocab_size, 128)

        # ÊΩúÂú®Á©∫Èñì„Å∏„ÅÆÂ∞ÑÂΩ±
        self.fc_mu = nn.Linear(256, latent_dim)       # Âπ≥ÂùáŒº
        self.fc_logvar = nn.Linear(256, latent_dim)   # log(ÂàÜÊï£)

        # DecoderÔºàÊΩúÂú®„Éô„ÇØ„Éà„É´ ‚Üí SMILESÔºâ
        self.decoder_fc = nn.Linear(latent_dim, 256)
        self.decoder_gru = nn.GRU(256, 256, num_layers=2, batch_first=True)
        self.decoder_output = nn.Linear(256, vocab_size)

    def encode(self, x):
        """
        Encoder: SMILES ‚Üí ÊΩúÂú®ÂàÜÂ∏ÉÔºàŒº„ÄÅœÉ¬≤Ôºâ

        Parameters:
        -----------

            SMILESÊñáÂ≠óÂàóÔºàÊï¥Êï∞„Ç®„É≥„Ç≥„Éº„Éá„Ç£„É≥„Ç∞Ôºâ

        --------

            ÊΩúÂú®ÂàÜÂ∏É„ÅÆÂπ≥Âùá
        logvar : torch.Tensor, shape (batch, latent_dim)
            ÊΩúÂú®ÂàÜÂ∏É„ÅÆlogÂàÜÊï£
        """

        embedded = self.encoder_embedding(x)  # (batch, max_length, 128)

        # GRUÔºàÊúÄÁµÇÈö†„ÇåÁä∂ÊÖã„Çí‰ΩøÁî®Ôºâ
        _, hidden = self.encoder_gru(embedded)  # hidden: (2, batch, 256)
        hidden = hidden[-1]  # ÊúÄÁµÇÂ±§„ÅÆÈö†„ÇåÁä∂ÊÖã: (batch, 256)

        # ÊΩúÂú®ÂàÜÂ∏É„ÅÆ„Éë„É©„É°„Éº„Çø
        mu = self.fc_mu(hidden)           # (batch, latent_dim)
        logvar = self.fc_logvar(hidden)   # (batch, latent_dim)

        return mu, logvar

    def reparameterize(self, mu, logvar):
        """
        Reparameterization Trick: z = Œº + œÉ * Œµ (Œµ ~ N(0, I))

        „Åì„Çå„Å´„Çà„Çä„ÄÅÁ¢∫ÁéáÁöÑ„Å™„Çµ„É≥„Éó„É™„É≥„Ç∞„ÇíÂãæÈÖçË®àÁÆóÂèØËÉΩ„Å´
        """
        std = torch.exp(0.5 * logvar)  # œÉ = exp(0.5 * log(œÉ¬≤))
        eps = torch.randn_like(std)    # Œµ ~ N(0, I)
        z = mu + std * eps             # z ~ N(Œº, œÉ¬≤)

        """
        Decoder: ÊΩúÂú®„Éô„ÇØ„Éà„É´ ‚Üí SMILES

        -----------

        max_length : int, optional
            ÁîüÊàê„Åô„ÇãSMILESÊúÄÂ§ßÈï∑ÔºàNone„ÅÆÂ†¥Âêà„ÅØself.max_lengthÔºâ

        """

        # ÊΩúÂú®„Éô„ÇØ„Éà„É´„ÇíÂàùÊúüÈö†„ÇåÁä∂ÊÖã„Å´Â§âÊèõ
        hidden = self.decoder_fc(z)  # (batch, 256)
        hidden = hidden.unsqueeze(0).repeat(2, 1, 1)  # (2, batch, 256)

        # „Éá„Ç≥„Éº„ÉÄ„Éº„ÅÆÂÖ•ÂäõÔºàÂÖ®ÊôÇÂàª„ÅßÊΩúÂú®„Éô„ÇØ„Éà„É´„ÇíÂÖ•ÂäõÔºâ
        decoder_input = z.unsqueeze(1).repeat(1, max_length, 1)  # (batch, max_length, latent_dim)

        output, _ = self.decoder_gru(decoder_input, hidden)  # (batch, max_length, 256)

        # ÂêÑ‰ΩçÁΩÆ„Åß„ÅÆÊñáÂ≠óÁ¢∫Áéá
        output = self.decoder_output(output)  # (batch, max_length, vocab_size)

        return output

    def forward(self, x):
        """
        È†Ü‰ºùÊí≠: x ‚Üí „Ç®„É≥„Ç≥„Éº„Éâ ‚Üí „Çµ„É≥„Éó„É™„É≥„Ç∞ ‚Üí „Éá„Ç≥„Éº„Éâ

        --------

        mu : ÊΩúÂú®ÂàÜÂ∏É„ÅÆÂπ≥Âùá

        """

        z = self.reparameterize(mu, logvar)
        recon_x = self.decode(z)
        return recon_x, mu, logvar

    def generate(self, num_samples=10, device='cpu'):
        """
        ÊΩúÂú®Á©∫Èñì„Åã„Çâ„É©„É≥„ÉÄ„É†„Çµ„É≥„Éó„É™„É≥„Ç∞„Åó„Å¶Êñ∞Ë¶èÂàÜÂ≠ê„ÇíÁîüÊàê

        Parameters:
        -----------
        num_samples : int
            ÁîüÊàê„Åô„ÇãÂàÜÂ≠êÊï∞
        device : str
            'cpu' or 'cuda'

        --------
        samples : torch.Tensor, shape (num_samples, max_length, vocab_size)
            ÁîüÊàê„Åï„Çå„ÅüÂàÜÂ≠êÔºàÁ¢∫ÁéáÂàÜÂ∏ÉÔºâ
        """
        self.eval()
        with torch.no_grad():
            # Ê®ôÊ∫ñÊ≠£Ë¶èÂàÜÂ∏É„Åã„Çâ„Çµ„É≥„Éó„É™„É≥„Ç∞

    # ÂÜçÊßãÊàêË™§Â∑ÆÔºàCross EntropyÔºâ
    recon_loss = F.cross_entropy(
        recon_x.view(-1, recon_x.size(-1)),  # (batch * max_length, vocab_size)
        x.view(-1),                          # (batch * max_length)
        reduction='sum'
    )

    # KL„ÉÄ„Ç§„Éê„Éº„Ç∏„Çß„É≥„Çπ: KL(N(Œº, œÉ¬≤) || N(0, I))
    # = -0.5 * Œ£(1 + log(œÉ¬≤) - Œº¬≤ - œÉ¬≤)
    kl_loss = -0.5 * torch.sum(1 + logvar - mu.pow(2) - logvar.exp())

    return recon_loss + kl_loss

        """

from rdkit import Chem
from rdkit.Chem import Descriptors, Lipinski
import pandas as pd

def check_drug_likeness(smiles):
    mol = Chem.MolFromSmiles(smiles)
    if mol is None:
        return False, {"error": "Invalid SMILES"}

    mw = Descriptors.MolWt(mol)
    logp = Descriptors.MolLogP(mol)
    hbd = Lipinski.NumHDonors(mol)
    hba = Lipinski.NumHAcceptors(mol)

    violations = 0
    if mw > 500: violations += 1
    if logp > 5: violations += 1
    if hbd > 5: violations += 1
    if hba > 10: violations += 1

    is_drug_like = violations <= 1

    properties = {
        'Compound': '',
        'Molecular Weight': round(mw, 2),
        'LogP': round(logp, 2),
        'H-Bond Donors': hbd,
        'H-Bond Acceptors': hba,
        'Lipinski Violations': violations,
        'Drug-like': is_drug_like
    }

    return is_drug_like, properties

    """
    2„Å§„ÅÆÂàÜÂ≠ê„ÅÆTanimotoÈ°û‰ººÂ∫¶„ÇíË®àÁÆó
