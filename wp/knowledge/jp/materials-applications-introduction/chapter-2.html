    Âá∫Âäõ: ÂÆπÈáè‰øùÊåÅÁéáÔºà%Ôºâat 5000„Çµ„Ç§„ÇØ„É´

    """

    def __init__(self, input_dim=5, hidden_dim=64, num_layers=2, dropout=0.2):
        super(BatteryDegradationLSTM, self).__init__()

        # LSTMÂ±§ÔºàÂèåÊñπÂêëÔºâ
        # ÂèåÊñπÂêë: ÈÅéÂéª„Å®Êú™Êù•„ÅÆÊÉÖÂ†±„Çí‰∏°Êñπ‰ΩøÁî®ÔºàË®ìÁ∑¥ÊôÇ„ÅÆ„ÅøÔºâ
        self.lstm = nn.LSTM(
            input_size=input_dim,
            hidden_size=hidden_dim,
            num_layers=num_layers,
            batch_first=True,
            dropout=dropout if num_layers > 1 else 0,
            bidirectional=True  # ÂèåÊñπÂêëLSTM
        )

        # Layer NormalizationÔºàË®ìÁ∑¥ÂÆâÂÆöÂåñÔºâ
        self.layer_norm = nn.LayerNorm(hidden_dim * 2)  # ÂèåÊñπÂêë„Å™„ÅÆ„Åß2ÂÄç

        self.dropout = nn.Dropout(dropout)

    def forward(self, x):
        """
        È†Ü‰ºùÊí≠

        # ÊúÄÂæå„ÅÆÊôÇÂàª„ÅÆ„Åø‰ΩøÁî®

        x = self.fc1(normalized)
        x = self.relu(x)
        x = self.dropout(x)

        # Âá∫ÂäõÂ±§
        capacity = self.fc2(x)  # (batch, 1)

        # SigmoidÔºà0-100%„Å´Âà∂ÈôêÔºâ
        capacity_retention = torch.sigmoid(capacity) * 100

        return capacity_retention

    Returns:
    --------
    sequences : np.ndarray, shape (n_batteries, n_cycles, 5)
        ÊôÇÁ≥ªÂàó„Éá„Éº„Çø
    labels : np.ndarray, shape (n_batteries,)
        5000„Çµ„Ç§„ÇØ„É´Âæå„ÅÆÂÆπÈáè‰øùÊåÅÁéáÔºà%Ôºâ
    """
    np.random.seed(42)
    sequences = []
    labels = []

    for i in range(n_batteries):
        # ÂàùÊúüÂÆπÈáè„ÅÆ„Å∞„Çâ„Å§„ÅçÔºàË£ΩÈÄ†Ë™§Â∑ÆÔºâ
        initial_capacity = np.random.normal(1.0, 0.02)  # Âπ≥Âùá1.0, Ê®ôÊ∫ñÂÅèÂ∑Æ0.02

        # Âä£ÂåñÈÄüÂ∫¶„ÅÆ„Å∞„Çâ„Å§„ÅçÔºàÊùêÊñôÂìÅË≥™„ÄÅ‰ΩøÁî®Êù°‰ª∂„ÅÆÈÅï„ÅÑÔºâ
        degradation_rate = np.random.uniform(0.00005, 0.0002)  # „Çµ„Ç§„ÇØ„É´„ÅÇ„Åü„Çä„ÅÆÂä£ÂåñÁéá

        # ÊôÇÁ≥ªÂàó„Éá„Éº„ÇøÁîüÊàê
        cycle_data = []
        for cycle in range(n_cycles):
            # ÂÆπÈáèÂä£ÂåñÔºàÊåáÊï∞Ê∏õË°∞Ôºâ
            capacity = initial_capacity * np.exp(-degradation_rate * cycle)

            # ÈõªÂúßÔºàÂÆπÈáè‰Ωé‰∏ã„Å´‰º¥„ÅÑÂ§âÂåñÔºâ
            voltage = 3.7 + 0.3 * capacity + np.random.normal(0, 0.01)

            # ÈõªÊµÅÔºàC-rate: 0.5-2.0Ôºâ
            current = np.random.uniform(0.5, 2.0)

            # Ê∏©Â∫¶Ôºà25¬±5¬∞CÔºâ
            temperature = 25 + np.random.normal(0, 5)

            # ÂÜÖÈÉ®ÊäµÊäóÔºàÂä£Âåñ„Å´‰º¥„ÅÑÂ¢óÂä†Ôºâ
            resistance = 0.05 + 0.01 * cycle / n_cycles + np.random.normal(0, 0.002)

            cycle_data.append([voltage, current, temperature, capacity, resistance])

        sequences.append(cycle_data)

        # 5000„Çµ„Ç§„ÇØ„É´Âæå„ÅÆÂÆπÈáè‰øùÊåÅÁéáÔºà„É©„Éô„É´Ôºâ
        final_capacity_retention = initial_capacity * np.exp(-degradation_rate * 5000) * 100
        labels.append(final_capacity_retention)

    return np.array(sequences), np.array(labels)

"""

class PerovskiteOptimizer:
    """
    „Éö„É≠„Éñ„Çπ„Ç´„Ç§„ÉàÂ§™ÈôΩÈõªÊ±†„ÅÆÁµÑÊàêÊúÄÈÅ©Âåñ

    - I: „É®„Ç¶Á¥†
    - Br: Ëá≠Á¥†

    ÊúÄÈÅ©Âåñ„Éë„É©„É°„Éº„Çø:
    - x: FA/MAÊØîÔºà0-1Ôºâ

    - annealing_time: ÁÑºÊàêÊôÇÈñìÔºà5-30ÂàÜÔºâ
    """

    def __init__(self):

        self.history = []

        # "Áúü„ÅÆ"ÊúÄÈÅ©Ëß£ÔºàÂÆüÈ®ì„Åß„ÅÆ„ÅøÂàÜ„Åã„ÇãÔºâ
        # ÂÆüÈöõ„ÅÆÂÆüÈ®ì„Åß„ÅØ„Åì„ÅÆÂÄ§„ÅØÊú™Áü•
        self.true_optimum = {

    def evaluate_performance(self, FA_ratio, I_ratio, annealing_temp, annealing_time):
        """
        ÂÆüÈ®ìÁöÑË©ï‰æ°ÔºàÂÖâÈõªÂ§âÊèõÂäπÁéá„ÅÆÊ∏¨ÂÆöÔºâ

        Returns:
        --------

            ÂÖâÈõªÂ§âÊèõÂäπÁéáÔºà%Ôºâ

        ------

        2. „Çπ„Éî„É≥„Ç≥„Éº„ÉÜ„Ç£„É≥„Ç∞
        3. ÁÑºÊàêÔºàÊ∏©Â∫¶„ÉªÊôÇÈñìÂà∂Âæ°Ôºâ

        fa_term = 20.0 * np.exp(-10 * (FA_ratio - self.true_optimum['FA_ratio'])**2)

        i_term = 5.0 * np.exp(-8 * (I_ratio - self.true_optimum['I_ratio'])**2)

        temp_deviation = (annealing_temp - self.true_optimum['annealing_temp']) / 20

        time_deviation = (annealing_time - self.true_optimum['annealing_time']) / 5
        time_term = 2.0 * np.exp(-time_deviation**2)

        baseline = 10.0

        efficiency = baseline + fa_term + i_term + temp_term + time_term + interaction

        noise = np.random.normal(0, 0.3)  # Ê®ôÊ∫ñÂÅèÂ∑Æ0.3%
        efficiency += noise

        # Áâ©ÁêÜÁöÑÂà∂Á¥ÑÔºà0-30%„ÅÆÁØÑÂõ≤Ôºâ
        efficiency = np.clip(efficiency, 0, 30)

        # ÂÆüÈ®ìÂ±•Ê≠¥„ÅÆË®òÈå≤
        self.history.append({
            'experiment': self.experimental_count,
            'FA_ratio': FA_ratio,
