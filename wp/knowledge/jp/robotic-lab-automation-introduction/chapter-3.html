from scipy.stats import norm

def expected_improvement(X, gp, f_best, xi=0.01):
    """
    Expected Improvementç²å¾—é–¢æ•°

    Args:
        X: å€™è£œç‚¹
        gp: ã‚¬ã‚¦ã‚¹éç¨‹ãƒ¢ãƒ‡ãƒ«
        f_best: ç¾åœ¨ã®æœ€è‰¯å€¤
        xi: Exploitation-Exploration trade-offï¼ˆå°ã•ã„ã»ã©æ´»ç”¨é‡è¦–ï¼‰

    Returns:
        EIå€¤
    """
    mu, sigma = gp.predict(X, return_std=True)

    # sigma=0ã®å ´åˆï¼ˆæ—¢ã«è¦³æ¸¬æ¸ˆã¿ï¼‰ã‚’å›é¿
    sigma = np.maximum(sigma, 1e-9)

    # EIè¨ˆç®—
    z = (mu - f_best - xi) / sigma
    ei = (mu - f_best - xi) * norm.cdf(z) + sigma * norm.pdf(z)

    return ei

        return best_result
